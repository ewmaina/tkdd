In this section, we present the definitions regarding graph and graph isomorphism. For convenience, we focus on undirected labeled graphs. 
The extension to other kinds of graphs is straightforward.

\begin{definition}
A \textit{labeled graph} is a six-element tuple $g= (V,E,L^v ,L^e ,\mu,\nu)$ where $V$ is a vertex set, $E$ is a edge set, both $\mu$ and $\nu$ 
are functions assigning labels to the vertices and edges respectively. $L^v$ and $L^e$ denotes the set of vertex and edge labels respectively.
\end{definition}

Following the convention, we denote a vertex set of graph $g$ as $V(g)$ and the edge set of graph $g$ as $E(g)$.

\begin{definition}
A graph $g_1=(V_1,E_1,L_1^v ,L_1^e ,\mu_1,\nu_1)$ is \textit{isomorphic} to another graph $g_2=(V_2,E_2,L_2^v ,L_2^e , \mu_2, \nu_2)$ if there exists a bijection $\phi$ such that
\begin{enumerate}
\item for every vertex $v \in V_1$, $\phi(v) \in V_2$ and $\mu_1(v) = \mu_2(\phi(v))$,
\item for every edge $(v,u) \in E_1$, $(\phi(v), \phi(u)) \in E_2$ and $\nu_1((u,v)) = \nu_2((\phi(v), \phi(u)))$
\end{enumerate}
\end{definition}

If $g_1$ is isomorphic to $g_2$, $\phi$ is called a \textit{graph isomorphism} and we use the notation $ g_1 \underset{iso}{=}  g_2$.

\begin{definition}
A graph $g_1=(V_1,E_1,L_1^v ,L_1^e ,\mu_1,\nu_1)$  is \textit{subgraph isomorphic} to another graph $g_2=(V_2,E_2,L_2^v ,L_2^e , \mu_2, \nu_2)$ if there exists a injection $\phi$ such that
\begin{enumerate}
\item for every vertex $v \in V_1$, $\phi(v) \in V_2$ and $\mu_1(v) = \mu_2(\phi(v))$,
\item for every edge $(v,u) \in E_1$, $(\phi(v), \phi(u)) \in E_2$ and $\nu_1((u,v)) = \nu_2((\phi(v), \phi(u)))$
\end{enumerate}
\end{definition}

If $g_1$ is subgraph isomorphic to $g_2$, $\phi$ is called a \textit{subgraph isomorphism},
$g_1$ is called a \textit{subgraph} of $g_2$ and we use the notation $g_1 \subseteq g_2$.

\begin{definition}
Given a graph $g_1=(V_1,E_1,L_1^{v} ,L_1^{e} ,\mu_1,\nu_1)$, a subgraph  $g_2=(V_2,E_2,L_2^{v} ,L_2^{e} , \mu_2, \nu_2)$ 
and a subgraph isomorphism $ \phi(\phi: V_2 \rightarrow V_1)$, $g_2$ is \textit{induced subgraph} of $g_1$ if it satisfies
\begin{itemize}
\item $ \exists (\phi_1(u), \phi_1(v)) \in E_1 \Rightarrow \exists (u,v) \in E_2$ for all $u,v \in V_2$
\end{itemize}
\end{definition}

If $g_2$ an is induced subgraph of $g_1$, $\phi$ is called an \textit{induced subgraph isomorphism} and we use the notation $\ g_{2} \underset{ind}{\subseteq} g_{1}$.

We now make a definitions and introduce the notation used in the processing of the graphs. Since we need to support both subgraphs and induced subgraphs, we provide a detailed description of the decomposition, union and subtraction of graphs for both these cases. Note that during random decomposition, the graphs are partitioned into roughly two equal number of vertices or edges for efficient processing. First we define the two cases of subgraph union, the union of two subgraphs with respect to a set of edges interconnecting them (induced subgraphs), and the union of two subgraphs with respect to a set of vertices common to both subgraphs (subgraphs).

\begin{definition}
\label{def:def3}
The union of two graphs $g_1=(V_1,E_1,L_1^{v} ,L_1^{e} ,\mu_1,\nu_1)$ and $g_2=(V_2,E_2,L_2^{v} ,L_2^{e} , \mu_2, \nu_2)$, where $V_{1} \cap V_{2} =\emptyset$, given a set of interconnecting edges $E^{'} \subseteq (V_{1} \times V_{2})$ with a labeling function $\nu:E^{'} \rightarrow L_{E}$, with respect to $E^{'}$ is the graph $g=(V,E,L^{v} ,L^{e} ,\mu,\nu)$ if:
\begin{enumerate}[1.]
\item $V=V_1 \cup V_2$
\item $E=E_1 \cup E_2 \cup E^{'}$
\item{ 
\[\mu(v) = \left\{
  \begin{array}{1 1}  
     \mu_{1}(v)  & \quad \text{if }  v \in V_{1} \\  
     \mu_{2}(v)  & \quad \text{if }  v \in V_{2} 
  \end{array} \right
\]
}

\item{ 
\[\nu(e) = \left\{
  \begin{array}{1 1}  \nu_{1}(e)  & \quad \text{if }  e \in E_{1} \\  
                      \nu_{2}(e)  & \quad \text{if } e \in E_{2} \\
                      \nu(e)     & \quad \text{if } e \in E^{'} 
  \end{array} \right
\]
}

\end{enumerate}
We denote this union of two graphs $g_1$ and $g_2$, with respect to a set of edges $E$ by $g_{1} \cup _{E} g_{2}$.
\end{definition}

\begin{definition}
\label{def:def31}
The union of two graphs $g_1=(V_1,E_1,L_1^{v} ,L_1^{e} ,\mu_1,\nu_1)$ and $g_2=(V_2,E_2,L_2^{v} ,L_2^{e} , \mu_2, \nu_2)$, where $E_{1} \cap E_{2} =\emptyset$, given a set of common vertices $V^{'} = V_{1} \cap V_{2}$ with respect to $V^{'}$ is the graph $g=(V,E,L^{v} ,L^{e} ,\mu,\nu)$ if:
\begin{enumerate}[1.]
\item $E=E_1 \cup E_2$
\item $V=V_1 \cup V_2 \cup V^{'}$
\item{ 
\[\mu(v) = \left\{
  \begin{array}{1 1}  
     \mu_{1}(v)  & \quad \text{if }  v \in V_{1} - V_{2}\\  
     \mu_{2}(v)  & \quad \text{if }  v \in V_{2} - V_{1}\\
     \mu(v)    & \quad \text{if }   v \in V_{1} \cap V_{2}
  \end{array} \right
\]
}

\item{ 
\[\nu(e) = \left\{
  \begin{array}{1 1}  \nu_{1}(e)  & \quad \text{if }  e \in E_{1} \\  
                      \nu_{2}(e)  & \quad \text{if } e \in E_{2} \\
  \end{array} \right
\]
}

\end{enumerate}
We denote this union of two graphs $g_1$ and $g_2$, with respect to a set of shared vertices $V$ by $g_{1} \cup _{V} g_{2}$.
\end{definition}


\begin{definition}
\label{def:def4}
A decomposition of a graph $g=(V,E,L^{v} ,L^{e} ,\mu,\nu)$ by partitioning of the set of vertices (also known as edge cutting), is a set of subgraphs $g_1=(V_1,E_1,L_1^{v} ,L_1^{e} ,\mu_1,\nu_1)$ and $g_2=(V_2,E_2,L_2^{v} ,L_2^{e} , \mu_2, \nu_2)$ that obeys the following conditions

\item $\mu_1 , \mu_2 , \nu_1, \nu_2$ are the restrictions of $\mu$ and $\nu$ to $V_1$, $V_2$ and $E_1$, $E_2$ respectively, that is:

Graph decomposition by partition of a set vertices (edge cutting) uniquely defines induced subgraphs of $g$ denoted by 
$g_{1} \underset{ind}{\subseteq} g$, $g_{2} \underset{ind}{\subseteq} g$ and satisfies $ g_{1} \cup_{E} g_{2} \underset{iso}{=} g$.    
\end{definition}


\begin{definition}
\label{def:def5}
The \textit{difference} between a graph $g$ and its induced subgraph $ g_1 \underset{ind}{\subseteq} g$ is the induced subgraph $g_2 \underset{ind}{\subseteq}  g$ that 
is defined by $ V_2 =V - V_1$.
We use the notation $ g\underset{ind}{-} g_1$ to represent the subtraction of induced subgraph.
\end{definition}
Note that both $g_1$ and $g_2$ are induced subgraphs of $g$ and there are edges that are  not included by either $g_1$ or $g_2$ as shown in definition \ref{def:def4}

\begin{definition}
\label{def:def7}
The \textit{difference} between a graph $g$ and a subgraph $g_1 \subseteq g$ is the subgraph $g_2 \subseteq g$ that is induced by $E_2 =E -E_1$.
We use the notation $g-g_1$ to represent the subtraction of subgraph.
\end{definition}
Note that both of $g_1$ and $g_2$ are subgraphs of $g$ and there are vertices both are included in both  $g_1$ and $g_2$  as shown in definition \ref{def:def6}.


\begin{definition}
Given a graph database $D= \{ g_1,g_2,\dots,g_n \}$ and a query graph $q$,
\textit{subgraph isomorphism query} detects all subgraph isomorphisms from $g \in D$ to $q$.
Similarly, \textit{induced subgraph isomorphism query} detects all induced subgraph isomorphisms from $g \in D$ to $q$.
\end{definition}

We call a graph that is present in the graph database a \textit{model graph}.
